// RAG Pipeline Configuration const RAG_CONFIG = { project: document.getElementById('projectSelect').value, embedding_model: 'text-embedding-3-small', llm_model: 'claude-sonnet-4-5-20250929', chunk_size: 400, chunk_overlap: 50, top_k: 10, similarity_threshold: 0.75, data_sources: { surveys: { sheet: '{project}_surveys', weight: 1.0 }, interviews: { sheet: '{project}_interviews', weight: 1.2 }, competitive: { sheet: '{project}_competitive', weight: 0.8 }, scoring: { sheet: '{project}_scoring', weight: 1.0 } }, metadata_filters: { date_range: 'last_90_days', min_relevance: 0.7, include_sources: true } }; // Step 1: Retrieve relevant chunks async function retrieveContext(query, config) { const embedding = await embed(query, config.embedding_model); const chunks = await vectorSearch(embedding, { top_k: config.top_k, threshold: config.similarity_threshold, filters: { project: config.project } }); return chunks; } // Step 2: Build prompt with retrieved context function buildPrompt(query, chunks) { const context = chunks.map((c, i) => `[Source ${i+1}: ${c.source_type} | ${c.date}]\n${c.text}` ).join('\n\n'); return `You are an expert product manager analyst. Answer the following question using ONLY the provided context. Cite sources using [Source N] notation. CONTEXT: ${context} QUESTION: ${query} Provide a structured, actionable answer with citations.`; } // Step 3: Generate answer async function generateAnswer(prompt, config) { const response = await llm.complete({ model: config.llm_model, prompt: prompt, max_tokens: 2000, temperature: 0.2 }); return response; } console.log('RAG Pipeline initialized for project:', RAG_CONFIG.project);

Query ResultCell 4

Query: "What are the top 3 pain points customers mentioned about dynamic pricing?"

Based on analysis of 9 survey responses and 4 interview transcripts, the top 3 pain points around dynamic pricing are:

1. Pricing Model Rigidity (mentioned by 7/9 respondents)
Customers report that current pricing systems cannot handle the complexity of hybrid models combining subscriptions, usage-based, and one-time charges. "Our SAP setup is too rigid for subscription-hybrid models" [Source 1: Interview, Caterpillar, Feb 20]. Survey data confirms 89% of respondents rated dynamic pricing as their top capability need [Source 3: Survey Aggregation].

2. Geo-Pricing Complexity (mentioned by 5/9 respondents)
Multi-currency and region-specific pricing rules are a major challenge. Companies operating globally need real-time pricing adjustments based on market conditions, tax jurisdictions, and competitive positioning [Source 2: Interview, Siemens, Feb 18]. 44% of survey respondents specifically requested geo-pricing capabilities [Source 4: Survey Data].

3. Volume Discount Management (mentioned by 4/9 respondents)
Enterprise customers struggle with tiered volume discounts that need to account for contract terms, commitment levels, and cross-product bundling. The current manual processes lead to revenue leakage estimated at 3-5% [Source 5: Interview, GE Digital, Feb 15].
